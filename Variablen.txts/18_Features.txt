let dlCount = 0;

function updateProgress() {

function renderGrid() {
  const grid = document.getElementById('grid');
  FILES.forEach(f => {
    const card = document.createElement('div');
    card.className = 'card';
    card.innerHTML = `
      <span class="badge ${f.cat}">${f.label}</span>
      <div class="title">${f.title}</div>
      <div class="desc">${f.desc}</div>
      <div class="files">${f.files}</div>
      <button class="dl-btn" id="btn-${f.id}" onclick="dlSingle(${f.id})">⬇️ Download</button>
    `;
    grid.appendChild(card);
  });
}

function dlSingle(id) {
  const f = FILES.find(x=>x.id===id);
  if(!f) return;
  const blob = new Blob([f.content],{type:'text/plain;charset=utf-8'});
  const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = 'atlas-feature-'+f.id+'-'+f.title.toLowerCase().replace(/[^a-z0-9]+/g,'-')+'.txt'; a.click();
  const btn = document.getElementById('btn-'+id); if(btn){btn.textContent='✓ Downloaded';btn.className='dl-btn done';}
  updateProgress();
}

function dlAll() {
  const all = FILES.map(f => [
    '='.repeat(80),
    'FEATURE '+f.id+' — '+f.title.toUpperCase()+' ['+f.label.toUpperCase()+']',
    '='.repeat(80),
    f.content,
    ''
  ].join('\n')).join('\n\n');

  const header = `================================================================================
ATLAS OS — ALLE 18+ FEHLENDEN FEATURES
AIAstralis · Version 20.0 · Complete Missing Features Bundle
Generated: ${new Date().toISOString()}
================================================================================

INHALT:
${FILES.map(f=>'Feature '+f.id+': '+f.title+' ['+f.label+']').join('\n')}

================================================================================


`;
  const blob = new Blob([header+all],{type:'text/plain;charset=utf-8'});
  const a = document.createElement('a'); a.href=URL.createObjectURL(blob); a.download='atlas-os-missing-18.txt'; a.click();
  FILES.forEach((_,i)=>setTimeout(()=>{
    const btn=document.getElementById('btn-'+(i+1)); if(btn){btn.textContent='✓ Downloaded';btn.className='dl-btn done';}
    updateProgress();
  },i*50));
