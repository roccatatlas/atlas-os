import { AFFILIATE_MAP, AFFILIATE_PROGRAMS } from "./affiliate-links";
import type { AffiliateProgram } from "./affiliate-links";

export function getAffiliateUrl(
  toolSlug: string,
  source: string = "tool-page",
  position: string = "cta"
): string | null {
  const prog = AFFILIATE_MAP[toolSlug];
  if (!prog) return null;
  const url = new URL(prog.baseUrl + prog.trackingId);
  url.searchParams.set("utm_source", "atlas-os");
  url.searchParams.set("utm_medium", "affiliate");
  url.searchParams.set("utm_campaign", source);
  url.searchParams.set("utm_content", position);
  return url.toString();
}

export async function trackClick(params: {
  toolSlug: string;
  source: string;
  position: string;
  keyword?: string;
}): Promise<void> {
  try {
    await fetch("/api/track", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(params),
    });
  } catch { /* silent fail */ }
}

export function estimateMonthlyRevenue(clicks: number, toolSlug: string): number {
  const prog = AFFILIATE_MAP[toolSlug];
  if (!prog) return 0;
  const conversions = clicks * (prog.convRate / 100);
  if (prog.commission.includes("CPA")) {
    const cpa = parseFloat(prog.commission.replace(/[^0-9.]/g, ""));
    return conversions * cpa;
  }
  if (prog.commission.includes("%")) {
    const pct = parseFloat(prog.commission);
    return conversions * 30 * (pct / 100);
  }
  return 0;
}

export function getTopAffiliates(limit = 10): AffiliateProgram[] {
  return AFFILIATE_PROGRAMS
    .sort((a, b) => (b.avgCPC * b.convRate) - (a.avgCPC * a.convRate))
    .slice(0, limit);
}

"use client";
import { getAffiliateUrl, trackClick } from "@/lib/affiliate";

type Props = {
  toolSlug: string;
  toolName: string;
  source?: string;
  position?: string;
  keyword?: string;
  size?: "sm" | "md" | "lg";
  variant?: "primary" | "outline" | "ghost";
  label?: string;
};

export function AffiliateBtn({
  toolSlug, toolName, source = "tool-page",
  position = "cta", keyword, size = "md",
  variant = "primary", label,
}: Props) {
  const url = getAffiliateUrl(toolSlug, source, position);

  const sizeClass = {
    sm: "px-3 py-1.5 text-xs",
    md: "px-5 py-2 text-sm",
    lg: "px-8 py-3 text-base",
  }[size];

  const variantClass = {
    primary: "bg-gradient-to-r from-cyan-400 to-blue-500 text-white shadow-[0_0_16px_#06b6d420] hover:opacity-90",
    outline: "border border-cyan-400/40 text-cyan-400 hover:border-cyan-400/80",
    ghost:   "text-cyan-400 hover:text-cyan-300",
  }[variant];

  async function handleClick() {
    await trackClick({ toolSlug, source, position, keyword });
    if (url) window.open(url, "_blank", "noopener,noreferrer");
  }

  if (!url) {
    return (
      <a href={"/tools/" + toolSlug}
        className={"inline-block rounded-lg font-bold transition-all " + sizeClass + " " + variantClass}>
        {label || "View " + toolName}
      </a>
    );
  }

  return (
    <button onClick={handleClick} rel="nofollow"
      className={"inline-block rounded-lg font-bold transition-all cursor-pointer " + sizeClass + " " + variantClass}>
      {label || "Try " + toolName + " Free →"}
    </button>
  );
}

import { NextRequest, NextResponse } from "next/server";
import { supabase } from "@/lib/supabase";

export async function POST(req: NextRequest) {
  const { toolSlug, source, position, keyword } = await req.json();
  const ip = req.headers.get("x-forwarded-for") || "unknown";
  const ipHash = Buffer.from(ip).toString("base64").slice(0, 16);

  await supabase.from("affiliate_clicks").upsert({
    tool_slug:  toolSlug,
    source,
    position,
    keyword:    keyword || null,
    ip_hash:    ipHash,
    clicked_at: new Date().toISOString(),
  });

  await supabase.rpc("increment_tool_clicks", { slug: toolSlug });

  return NextResponse.json({ ok: true });
}

// FILE: app/api/revenue/route.ts

import { NextResponse } from "next/server";
import { supabase } from "@/lib/supabase";
import { AFFILIATE_PROGRAMS } from "@/lib/affiliate-links";
import { estimateMonthlyRevenue } from "@/lib/affiliate";

export async function GET() {
  const { data: clicks } = await supabase
    .from("affiliate_clicks")
    .select("tool_slug")
    .gte("clicked_at", new Date(Date.now() - 30 * 86400000).toISOString());

  const clickMap: Record<string, number> = {};
  clicks?.forEach(c => {
    clickMap[c.tool_slug] = (clickMap[c.tool_slug] || 0) + 1;
  });

  const revenue = AFFILIATE_PROGRAMS.map(prog => ({
    toolSlug:         prog.toolSlug,
    toolName:         prog.toolName,
    category:         prog.category,
    commission:       prog.commission,
    clicks:           clickMap[prog.toolSlug] || 0,
    estimatedRevenue: estimateMonthlyRevenue(clickMap[prog.toolSlug] || 0, prog.toolSlug),
    convRate:         prog.convRate,
    network:          prog.network,
  })).sort((a, b) => b.estimatedRevenue - a.estimatedRevenue);

  const totalRevenue = revenue.reduce((s, r) => s + r.estimatedRevenue, 0);
  const totalClicks  = Object.values(clickMap).reduce((s, v) => s + v, 0);

  return NextResponse.json({ revenue, totalRevenue, totalClicks });
}
