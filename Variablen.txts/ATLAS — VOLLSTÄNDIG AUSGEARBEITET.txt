A-01. DISCOVERY SYSTEM
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ZWECK:
  Findet automatisch neue AI-Tools aus dem Internet und speichert sie in der DB.

KOMPONENTEN:
  → AI Tool Discovery Engine    Crawlt Product Hunt, GitHub, AI Directories
  → Startup Discovery Engine    Scannt AngelList, Crunchbase, YC, TechCrunch
  → Open Source AI Scanner      Überwacht GitHub Topics: ai-tools, llm, agents
  → AI Research Paper Scanner   Scannt ArXiv, Papers With Code für neue Modelle

QUELLEN:
  Product Hunt, GitHub Topics, Futurepedia, TheresAnAIForThat,
  HackerNews, AI Newsletters, AngelList, Crunchbase

FLOW:
  Internet → Source Collectors → Web Crawler → Tool Detector
  → Metadata Extractor → AI Classification → Deduplication → Database

OUTPUT PRO TOOL:
  tool_name, website, description, tags, category, source, discovered_at

TECHNOLOGIE:
  Python, Scrapy / Requests / BeautifulSoup,
  Celery (Scheduling), Redis (Queue), PostgreSQL (Storage)

SCHEDULER:
  Celery Beat — läuft alle 6 Stunden automatisch

SKALIERUNG:
  Ziel: 1.000+ neue Tools pro Monat automatisch erfasst

WAS NOCH FEHLT (MVP):
  Scheduler, Deduplication, Domain Parsing, Error Handling

ZWECK:

KOMPONENTEN:
  → AI Tool Knowledge Graph     Nodes: Tools, Models, Agents, Frameworks
                                Edges: integration, workflow, competitor, category

  → AI Integration Database     Welche Tools können miteinander verbunden werden

GRAPH-STRUKTUR:
  Nodes    = AI Tools (id, name, slug, category, description)
  Edges    = Relationen (integration, workflow_step, comparison, alternative)
  Clusters = Kategorien (LLMs, Image Gen, Automation, Coding, etc.)

  Optional: Neo4j für komplexere Graph-Queries,
  pgvector für semantische Embeddings

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ZWECK:

  extrahiert strukturierte Metadaten.

KOMPONENTEN:
  → AI Tool Analyzer            GPT-gestützte Tool-Analyse via Webscraping
  → Workflow Detection Engine   Erkennt Workflows aus Integration-Graphen

  → AI Capability Classifier    Kategorisiert: text, image, video, code, audio...

AI ANALYZER FLOW:
  Tool URL → Scrape Website → GPT Prompt → Structured Output → DB

GPT PROMPT TEMPLATE:
  "Analyze the AI tool [NAME] at [URL].
   Return JSON with:
   - category (string)
   - features (array)
   - integrations (array)
   - use_cases (array)
   - pricing_model (free/freemium/paid/enterprise)
   - api_available (boolean)
   - target_audience (string)"

OUTPUT:
  category, features[], integrations[], use_cases[],
  pricing_model, api_available, target_audience

TECHNOLOGIE:
  Python, OpenAI API (GPT-4), BeautifulSoup, psycopg2

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  User-Intent, Graph-Relationen und Popularität.

KOMPONENTEN:
  → Tool Recommendation Engine      Empfiehlt einzelne Tools
  → Workflow Recommendation Engine  Empfiehlt komplette Workflows
  → AI Stack Recommender            Empfiehlt Stack für einen Use Case
  → Integration Recommender         Empfiehlt kompatible Tools

SCORING-ALGORITHMUS:
  score = integration_score
        + workflow_score
        + trend_score
        + popularity_score

  integration_score  = Anzahl gemeinsamer Integrations-Edges
  workflow_score     = Häufigkeit in gespeicherten Workflows
  trend_score        = GitHub Stars Growth + Product Hunt Votes
  popularity_score   = Klicks + Stack-Appearances

D-02. TOOL RELATIONSHIP ENGINE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ZWECK:
  Erfasst welche Tools häufig gemeinsam in Stacks vorkommen.
  Das ist der eigentliche MOAT von Atlas.

MECHANISMUS:
  Jedes Mal wenn ein Stack gespeichert wird →
  alle Tool-Paare extrahieren → co_occurrence Tabelle updaten

MOAT-LOGIK:
  Je mehr Stacks gespeichert werden, desto besser die Daten.
  Diese Daten hat kein anderes Tool-Directory.
  → Atlas wird zur einzigen Quelle für "welche Tools gehören zusammen"

SQL BEISPIEL:
  INSERT INTO tool_co_occurrence (tool_a_id, tool_b_id, count)
  VALUES (?, ?, 1)
  ON CONFLICT (tool_a_id, tool_b_id) DO UPDATE SET count = count + 1;

[DB SCHEMA — POSTGRESQL CORE]
──────────────────────────────
CREATE TABLE categories (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  slug VARCHAR(255) UNIQUE,
  description TEXT
);

CREATE TABLE tools (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  slug VARCHAR(255) UNIQUE,
  website TEXT,
  description TEXT,
  category_id INT REFERENCES categories(id),
  pricing_model VARCHAR(100),
  api_available BOOLEAN,
  logo_url TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE tool_relations (
  id SERIAL PRIMARY KEY,
  source_tool INT REFERENCES tools(id),
  target_tool INT REFERENCES tools(id),
  relation_type VARCHAR(100)
);

CREATE TABLE stacks (
  id SERIAL PRIMARY KEY,
  title VARCHAR(255),
  slug VARCHAR(255) UNIQUE,
  description TEXT,
  prompt TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE stack_steps (
  id SERIAL PRIMARY KEY,
  stack_id INT REFERENCES stacks(id),
  tool_id INT REFERENCES tools(id),
  step_order INT,

def discover_tools(url):
    response = requests.get(url, timeout=10)
    soup = BeautifulSoup(response.text, "html.parser")
    tools = []
    for link in soup.find_all("a"):
        name = link.text.strip()
        href = link.get("href")
        if name and href and "http" in href and len(name) > 2:
            tools.append({"name": name, "website": href})
    return tools

def detect_workflows(graph):
    """graph = { tool_id: [connected_tool_ids] }"""
    workflows = []
    for node in graph:
        for neighbor in graph.get(node, []):
            for next_node in graph.get(neighbor, []):
                if next_node != node:
                    workflows.append([node, neighbor, next_node])
    return workflows

// app/api/v1/generate-stack/route.ts
import OpenAI from "openai"
import { prisma } from "@/lib/prisma"

const openai = new OpenAI()

export async function POST(req: Request) {
  const { prompt } = await req.json()
  const tools = await prisma.tool.findMany({
    select: { id: true, name: true, slug: true, description: true }
  })
  const toolList = tools.map(t => `${t.name}: ${t.description}`).join("\n")

  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{
      role: "user",
      content: `User wants to: "${prompt}"\n\nAvailable tools:\n${toolList}\n\n
      Return a JSON array of steps:
      [{"step": 1, "tool_name": "...", "action": "...", "reason": "..."}]
      Only use tools from the list above. Max 6 steps.`
    }],
    response_format: { type: "json_object" }
  })

  const steps = JSON.parse(response.choices[0].message.content || "{}")
  return Response.json({ steps })
}
