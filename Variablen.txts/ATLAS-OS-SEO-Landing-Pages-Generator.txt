import { supabase } from "./supabase";
import type { Tool } from "./supabase";

export interface SEOPageMeta {
  title: string;
  description: string;
  h1: string;
  intro: string;
  faqs: { q: string; a: string }[];
  schema: object;
}

export async function getToolsForKeyword(keyword: string): Promise<Tool[]> {
  const catMap: Record<string, string> = {
    writing: "Writing", coding: "Coding", image: "Image",
    video: "Video", audio: "Audio", research: "Research",
    automation: "Automation", productivity: "Productivity",
    design: "Design", marketing: "Marketing",
  };

  let category: string | null = null;
  for (const [k, v] of Object.entries(catMap)) {
    if (keyword.includes(k)) { category = v; break; }
  }

  let q = supabase.from("tools").select("*").eq("status","active")
    .order("atlas_score", { ascending: false }).limit(10);

  if (category) q = q.eq("category", category);

  const { data } = await q;
  return data || [];
}

import { generateKeywords, keywordToSlug } from "../lib/keywords";
import { supabase } from "../lib/supabase";
import fs from "fs";

async function main() {
  const keywords = generateKeywords();
  console.log("ATLAS SEO Generator");
  console.log("Generating " + keywords.length + " keyword pages...");

  const pages = keywords.map(kw => ({
    path:           "/best-ai-tools-for/" + keywordToSlug(kw),
    page_type:      "keyword",
    indexing_state: "noindex",
    canonical_url:  process.env.NEXT_PUBLIC_SITE_URL + "/best-ai-tools-for/" + keywordToSlug(kw),
  }));

  const CHUNK = 500;
  for (let i = 0; i < pages.length; i += CHUNK) {
    const chunk = pages.slice(i, i + CHUNK);
    const { error } = await supabase.from("seo_pages").upsert(chunk, { onConflict: "path" });
    if (error) console.error("Chunk " + i + " error:", error.message);
    else console.log("Inserted pages " + i + " to " + (i + chunk.length));
  }

  const sitemap = pages.map(p =>
    "  <url><loc>" + p.canonical_url + "</loc><changefreq>weekly</changefreq><priority>0.7</priority></url>"
  ).join("\\n");

  fs.writeFileSync("public/sitemap-keywords.xml",
    '<?xml version="1.0" encoding="UTF-8"?>\\n<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\\n' + sitemap + '\\n</urlset>'
  );

  console.log("Done. " + pages.length + " pages registered.");
  console.log("Sitemap written to public/sitemap-keywords.xml");
}

main().catch(console.error);
