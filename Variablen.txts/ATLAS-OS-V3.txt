// ── ANIMATED COUNTER ──────────────────────────────────────────
function Counter({end,suffix=""}) {
  const [val,setVal]=useState(0);
  useEffect(()=>{
    let start=0; const step=end/60;
    const t=setInterval(()=>{ start+=step; if(start>=end){setVal(end);clearInterval(t);}else setVal(Math.floor(start)); },16);
    return ()=>clearInterval(t);
  },[end]);
  return <span>{val.toLocaleString()}{suffix}</span>;
}

// ── TYPEWRITER ────────────────────────────────────────────────
function Typewriter({texts}) {
  const [idx,setIdx]=useState(0);
  const [shown,setShown]=useState("");
  const [del,setDel]=useState(false);
  useEffect(()=>{
    const t=texts[idx];
    if(!del&&shown.length<t.length){ const id=setTimeout(()=>setShown(t.slice(0,shown.length+1)),40); return()=>clearTimeout(id); }
    if(!del&&shown.length===t.length){ const id=setTimeout(()=>setDel(true),1800); return()=>clearTimeout(id); }
    if(del&&shown.length>0){ const id=setTimeout(()=>setShown(shown.slice(0,-1)),20); return()=>clearTimeout(id); }
    if(del&&shown.length===0){ setDel(false); setIdx((idx+1)%texts.length); }
  },[shown,del,idx,texts]);
  return <span style={{color:"#06b6d4"}}>{shown}<span style={{animation:"blink 1s infinite",opacity:.8}}>|</span></span>;
}

// ── COMPARE ───────────────────────────────────────────────────
function Compare({initA,initB}) {
  const [a,setA]=useState(initA||null);
  const [b,setB]=useState(initB||null);
  const [pick,setPick]=useState(null);

  function sel(t){ if(pick==="a"){setA(t);setPick(null);}else if(pick==="b"){setB(t);setPick(null);} }

  const ROWS = [
    {label:"Company",key:"company"},
    {label:"Category",key:"cat"},
    {label:"Pricing",key:"pricing"},
    {label:"Rating",key:"rating",render:(v)=><Stars n={v}/>},
    {label:"Users",key:"users"},
    {label:"API Access",key:"api",render:(v)=>v?<Badge color="#22c55e" sm>✓ Yes</Badge>:<Badge color="#ef4444" sm>✗ No</Badge>},
    {label:"Open Source",key:"openSource",render:(v)=>v?<Badge color="#f59e0b" sm>✓ Yes</Badge>:<Badge color="#475569" sm>✗ No</Badge>},
    {label:"Deployment",key:"deploy"},
    {label:"Context Window",key:"contextWindow"},
    {label:"Integrations",key:"integrations",render:(v)=>`${v}+`},
  ];

  function winner(attr,ta,tb) {
    if(attr.key==="rating") return ta.rating>tb.rating?"a":tb.rating>ta.rating?"b":"tie";
    if(attr.key==="integrations") return ta.integrations>tb.integrations?"a":tb.integrations>ta.integrations?"b":"tie";
    if(attr.key==="api") return ta.api&&!tb.api?"a":!ta.api&&tb.api?"b":"tie";
    return "tie";
  }

  const winsA = a&&b ? ROWS.filter(r=>winner(r,a,b)==="a").length : 0;
  const winsB = a&&b ? ROWS.filter(r=>winner(r,a,b)==="b").length : 0;

  return (
    <section style={{maxWidth:900,margin:"0 auto",padding:"52px 20px"}}>
      <Badge>Comparison Engine · Part 23</Badge>
      <h2 style={{fontSize:32,fontWeight:900,margin:"14px 0 6px",color:"#f1f5f9"}}>Compare AI Tools</h2>
      <p style={{color:"#475569",marginBottom:28,fontSize:13}}>Side-by-side comparison across 10 attributes with ATLAS Verdict.</p>

      {/* Selectors */}
      <div style={{display:"grid",gridTemplateColumns:"1fr 50px 1fr",gap:10,marginBottom:20,alignItems:"center"}}>
        {[["a",a],["b",b]].map(([side,tool])=>(
          <div key={side} onClick={()=>setPick(side)} style={{background:pick===side?"#06b6d412":"#0f172a",border:`2px solid ${pick===side?"#06b6d4":tool?"#1e3a5f":"#1e293b"}`,borderRadius:12,padding:16,cursor:"pointer",textAlign:"center",minHeight:88,display:"flex",flexDirection:"column",alignItems:"center",justifyContent:"center",transition:"all .2s"}}
            onMouseEnter={e=>e.currentTarget.style.borderColor="#06b6d444"}
            onMouseLeave={e=>e.currentTarget.style.borderColor=pick===side?"#06b6d4":tool?"#1e3a5f":"#1e293b"}>
            {tool?(<><div style={{fontSize:26,marginBottom:4}}>{tool.logo}</div><div style={{fontWeight:700,color:"#f1f5f9",fontSize:13}}>{tool.name}</div><div style={{fontSize:10,color:"#475569",marginTop:2}}>{tool.cat}</div></>):(<div style={{color:pick===side?"#06b6d4":"#334155",fontSize:12}}>{pick===side?"Click a tool below":"+ Select Tool"}</div>)}
          </div>
        ))}
        <div style={{textAlign:"center",color:"#1e293b",fontSize:18,fontWeight:900}}>VS</div>
      </div>

      {/* Tool picker */}
      {pick && (
        <div style={{background:"#0f172a",border:"1px solid #06b6d433",borderRadius:12,padding:14,marginBottom:20,animation:"fadeIn .2s ease"}}>
          <div style={{fontSize:11,color:"#06b6d4",marginBottom:10,fontWeight:700}}>SELECT TOOL {pick.toUpperCase()}:</div>
          <div style={{display:"flex",flexWrap:"wrap",gap:6}}>
            {TOOLS.map(t=>(
              <button key={t.id} onClick={()=>sel(t)} style={{background:"#1e293b",border:"1px solid #334155",color:"#94a3b8",borderRadius:8,padding:"5px 10px",fontSize:11,cursor:"pointer",display:"flex",alignItems:"center",gap:5,transition:"all .15s"}}
                onMouseEnter={e=>{e.currentTarget.style.background="#243040";e.currentTarget.style.color="#f1f5f9";}}
                onMouseLeave={e=>{e.currentTarget.style.background="#1e293b";e.currentTarget.style.color="#94a3b8";}}>
                {t.logo} {t.name}
              </button>
            ))}
          </div>
        </div>
      )}

      {/* Score bar */}
      {a&&b&&(
        <div style={{background:"#0f172a",border:"1px solid #1e293b",borderRadius:10,padding:"12px 16px",marginBottom:16,display:"flex",gap:12,alignItems:"center"}}>
          <span style={{fontSize:13,fontWeight:700,color:"#f1f5f9",minWidth:90}}>{a.name}: {winsA}</span>
          <div style={{flex:1,height:6,background:"#1e293b",borderRadius:3,overflow:"hidden"}}>
            <div style={{height:"100%",background:"linear-gradient(90deg,#06b6d4,#3b82f6)",width:`${winsA/(winsA+winsB||1)*100}%`,transition:"width .5s ease",borderRadius:3}}/>
          </div>
          <span style={{fontSize:13,fontWeight:700,color:"#f1f5f9",minWidth:90,textAlign:"right"}}>{b.name}: {winsB}</span>
        </div>
      )}

      {/* Table */}
      {a&&b&&(
        <div style={{background:"#0f172a",border:"1px solid #1e293b",borderRadius:14,overflow:"hidden"}}>
          <div style={{display:"grid",gridTemplateColumns:"140px 1fr 1fr",background:"#080f1e"}}>
            <div style={{padding:"12px 14px",fontSize:10,color:"#334155",fontWeight:700,letterSpacing:.5,borderRight:"1px solid #1e293b"}}>ATTRIBUTE</div>
            {[a,b].map((t,i)=>(
              <div key={i} style={{padding:"12px 14px",textAlign:"center",borderRight:i===0?"1px solid #1e293b":"none"}}>
                <div style={{fontSize:20,marginBottom:2}}>{t.logo}</div>
                <div style={{fontWeight:800,color:"#f1f5f9",fontSize:12}}>{t.name}</div>
              </div>
            ))}
          </div>
          {ROWS.map((row,i)=>{
            const w=winner(row,a,b);
            return (
              <div key={row.key} style={{display:"grid",gridTemplateColumns:"140px 1fr 1fr",background:i%2===0?"#0f172a":"#0a111e",borderTop:"1px solid #0d1b2e"}}>
                <div style={{padding:"11px 14px",fontSize:11,color:"#475569",fontWeight:600,borderRight:"1px solid #0d1b2e",display:"flex",alignItems:"center"}}>{row.label}</div>
                {[a,b].map((t,j)=>(
                  <div key={j} style={{padding:"11px 14px",textAlign:"center",fontSize:11,color:"#94a3b8",borderRight:j===0?"1px solid #0d1b2e":"none",display:"flex",alignItems:"center",justifyContent:"center",gap:4,background:(w==="a"&&j===0)||(w==="b"&&j===1)?"#06b6d408":"transparent",borderBottom:(w==="a"&&j===0)||(w==="b"&&j===1)?"2px solid #06b6d422":"2px solid transparent"}}>
                    {(w==="a"&&j===0)||(w==="b"&&j===1)?<span style={{color:"#06b6d4",fontSize:9}}>▲</span>:null}
                    {row.render?row.render(t[row.key]):<span style={{color:"#94a3b8"}}>{t[row.key]}</span>}
                  </div>
                ))}
              </div>
            );
          })}
          {/* Verdict */}
          <div style={{padding:"16px 18px",background:"#06b6d408",borderTop:"1px solid #06b6d422"}}>
            <div style={{fontWeight:700,color:"#06b6d4",fontSize:12,marginBottom:6}}>⚡ ATLAS Verdict</div>
            <div style={{fontSize:12,color:"#94a3b8",lineHeight:1.75}}>
              {winsA>winsB
                ?<><strong style={{color:"#f1f5f9"}}>{a.name}</strong> wins on {winsA} of {ROWS.length} attributes. Best for: {a.cat.toLowerCase()} workflows{a.api?", API integration":""}{a.openSource?", open-source flexibility":""}.</>
                :winsB>winsA
                ?<><strong style={{color:"#f1f5f9"}}>{b.name}</strong> wins on {winsB} of {ROWS.length} attributes. Best for: {b.cat.toLowerCase()} workflows{b.api?", API integration":""}{b.openSource?", open-source flexibility":""}.</>
                :<>It's a tie at {winsA} each. Choose based on your specific use case and budget.</>
              }
            </div>
            <div style={{marginTop:10,display:"flex",gap:6}}>
              <Btn sm>Add to Stack</Btn>
              <Btn sm outline onClick={()=>{setA(null);setB(null);}}>Reset</Btn>
            </div>
          </div>
        </div>
      )}

      {(!a||!b)&&!pick&&(
        <div style={{textAlign:"center",padding:48,color:"#1e293b",fontSize:13,border:"1px dashed #1e293b",borderRadius:14}}>
          Click the slots above to select two tools to compare
        </div>
      )}
      <style>{`@keyframes fadeIn{from{opacity:0;transform:translateY(6px)}to{opacity:1;transform:translateY(0)}}`}</style>
    </section>
  );
}
