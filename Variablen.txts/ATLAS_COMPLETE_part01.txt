function getAvatarColor(name: string): string {
  const colors = [
    "#4DA3FF",
    "#2EC4B6",
    "#E040FB",
    "#FF9100",
    "#69F0AE",
    "#FF5252",
    "#7C4DFF",
    "#FFAB40",
  ];
  let hash = 0;
  for (let i = 0; i < name.length; i++) {
    hash = name.charCodeAt(i) + ((hash << 5) - hash);
  }
  return colors[Math.abs(hash) % colors.length];
}

function TiltCard({ children, className, ...props }: React.HTMLAttributes<HTMLDivElement>) {
  const cardRef = useRef<HTMLDivElement>(null);

  const handleMouseMove = useCallback((e: React.MouseEvent) => {
    const card = cardRef.current;
    if (!card) return;
    const rect = card.getBoundingClientRect();
    const x = (e.clientX - rect.left) / rect.width;
    const y = (e.clientY - rect.top) / rect.height;
    const tiltX = (y - 0.5) * -8;
    const tiltY = (x - 0.5) * 8;
    card.style.transform = `perspective(600px) rotateX(${tiltX}deg) rotateY(${tiltY}deg) translateY(-4px)`;
    card.style.setProperty('--mouse-x', `${x * 100}%`);
    card.style.setProperty('--mouse-y', `${y * 100}%`);
  }, []);

  const handleMouseLeave = useCallback(() => {
    const card = cardRef.current;
    if (card) {
      card.style.transform = 'perspective(600px) rotateX(0deg) rotateY(0deg) translateY(0px)';
    }
  }, []);

  return (
    <div
      ref={cardRef}
      className={`tilt-card ${className}`}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
      style={{ transition: 'transform 0.15s ease-out' }}
      {...props}
    >
      <div className="tilt-card-shine" />
      {children}
    </div>
  );
}

function StarRating({ rating }: { rating: number }) {
  return (
    <div className="flex items-center gap-1" aria-label={`${rating} von 5 Sternen`}>
      {Array.from({ length: 5 }, (_, i) => {
        const filled = i < Math.floor(rating);
        const half = !filled && i < rating;
        return (
          <span key={i} className="relative">
            <Star
              className={`w-5 h-5 ${
                filled
                  ? "text-yellow-400 fill-yellow-400"
                  : half
                  ? "text-yellow-400"
                  : "text-white/20"
              }`}
            />
            {half && (
              <span className="absolute inset-0 overflow-hidden w-[50%]">
                <Star className="w-5 h-5 text-yellow-400 fill-yellow-400" />
              </span>
            )}
          </span>
        );
      })}
      <span className="ml-2 text-sm text-white/60 font-medium">{rating.toFixed(1)}</span>
    </div>
  );
}

function polarToCartesian(
  cx: number,
  cy: number,
  radius: number,
  angleIndex: number,
  total: number
): [number, number] {
  const angle = (2 * Math.PI * angleIndex) / total - Math.PI / 2;
  return [cx + radius * Math.cos(angle), cy + radius * Math.sin(angle)];
}

function buildPolygonPoints(
  cx: number,
  cy: number,
  maxRadius: number,
  values: number[]
): string {
  return values
    .map((v, i) => {
      const r = (v / 100) * maxRadius;
      const [x, y] = polarToCartesian(cx, cy, r, i, values.length);
      return `${x},${y}`;
    })
    .join(" ");
}

function buildGridPolygon(
  cx: number,
  cy: number,
  radius: number,
  n: number
): string {
  return Array.from({ length: n }, (_, i) => {
    const [x, y] = polarToCartesian(cx, cy, radius, i, n);
    return `${x},${y}`;
  }).join(" ");
}

  description: string;

  pricing: 'Free' | 'Freemium' | 'Paid';
  tags: string[];
  url: string;

  sponsored?: boolean;
}

export interface BlogPost {
  slug: string;
  title: string;
  excerpt: string;
  date: string;
  author: string;
  image?: string;
  tags: string[];
}

// Data fetcher abstraction
const API_BASE = import.meta.env.VITE_API_URL || '';
const USE_API = !!API_BASE;

async function fetchFromApi<T>(endpoint: string): Promise<T> {
  const res = await fetch(`${API_BASE}${endpoint}`);
  if (!res.ok) throw new Error(`API error: ${res.status}`);
  return res.json();
}

// Static data imports (tree-shaken when API is used)
async function getStaticTools(): Promise<ToolData[]> {
  const { TOOLS_DATA } = await import('@/data/toolsData');
  return TOOLS_DATA;
}

async function getStaticBlogPosts(): Promise<BlogPost[]> {
  const { BLOG_POSTS } = await import('@/data/blogData');
  return BLOG_POSTS;
}

// Public API
export const toolsApi = {
  getTools: (): Promise<ToolData[]> =>
    USE_API ? fetchFromApi<ToolData[]>('/api/tools') : getStaticTools(),

  getToolById: async (id: string): Promise<ToolData | undefined> => {
    const tools = await toolsApi.getTools();
    return tools.find(t => t.id === id);
  },

  getCategories: async (): Promise<string[]> => {
    const tools = await toolsApi.getTools();
    return [...new Set(tools.map(t => t.category))].sort();
  },

  getTrending: async (limit = 10): Promise<ToolData[]> => {
    const tools = await toolsApi.getTools();
    return tools.sort((a, b) => b.rating - a.rating).slice(0, limit);
  },

  search: async (query: string): Promise<ToolData[]> => {
    const tools = await toolsApi.getTools();
    const q = query.toLowerCase();
    return tools.filter(t =>
      t.name.toLowerCase().includes(q) ||
      t.description.toLowerCase().includes(q) ||
      t.tags.some(tag => tag.includes(q))
    );
  },
};

export const blogApi = {
  getPosts: (): Promise<BlogPost[]> =>
    USE_API ? fetchFromApi<BlogPost[]>('/api/blog') : getStaticBlogPosts(),

  getPost: async (slug: string): Promise<BlogPost | undefined> => {
    const posts = await blogApi.getPosts();
    return posts.find(p => p.slug === slug);
  },
};

export function generateOGImageUrl(params: {
  title: string;
  subtitle?: string;
  category?: string;
  toolCount?: number;
}): string {
  const base = 'https://runatlas.com/api/og';
  const searchParams = new URLSearchParams();
  searchParams.set('title', params.title);
  if (params.subtitle) searchParams.set('subtitle', params.subtitle);
  if (params.category) searchParams.set('category', params.category);
  if (params.toolCount) searchParams.set('count', String(params.toolCount));
  return `${base}?${searchParams.toString()}`;
}

// Lightweight client-side analytics — stores events locally and optionally sends to backend
interface AnalyticsEvent {
  name: string;
  properties?: Record<string, string | number | boolean>;
  timestamp: number;
}

const STORAGE_KEY = 'runatlas-analytics';
const MAX_EVENTS = 500;

function getEvents(): AnalyticsEvent[] {
  try {
    return JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]');
  } catch { return []; }
}

function saveEvents(events: AnalyticsEvent[]) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(events.slice(-MAX_EVENTS)));
  } catch {}
}

export function trackEvent(name: string, properties?: Record<string, string | number | boolean>) {
  const event: AnalyticsEvent = { name, properties, timestamp: Date.now() };
  const events = [...getEvents(), event];
  saveEvents(events);

  // Forward to GTM / custom backend if available
  if (typeof window.dataLayer !== 'undefined') {
    window.dataLayer.push({ event: name, ...properties });
  }

  if (import.meta.env.DEV) {
    console.log(`[Analytics] ${name}`, properties);
  }
}

export function getAnalyticsSummary() {
  const events = getEvents();
  const now = Date.now();
  const last24h = events.filter(e => now - e.timestamp < 86400000);
  const last7d = events.filter(e => now - e.timestamp < 604800000);

  // Count by event name
  const counts: Record<string, number> = {};
  events.forEach(e => { counts[e.name] = (counts[e.name] || 0) + 1; });

  // Most clicked tools
  const toolClicks = events
    .filter(e => e.name === 'tool_click' && e.properties?.tool)
    .reduce<Record<string, number>>((acc, e) => {
      const tool = String(e.properties!.tool);
      acc[tool] = (acc[tool] || 0) + 1;
      return acc;
    }, {});

  const topTools = Object.entries(toolClicks)
    .sort(([, a], [, b]) => b - a)
    .slice(0, 10);

  return {
    totalEvents: events.length,
    last24h: last24h.length,
    last7d: last7d.length,
    eventCounts: counts,
    topTools,
  };
}

// Utility to wrap tool URLs with UTM + affiliate parameters
interface AffiliateConfig {
  utmSource: string;
  utmMedium: string;
  utmCampaign: string;
  affiliateId?: string;
}

const DEFAULT_CONFIG: AffiliateConfig = {
  utmSource: 'runatlas',
  utmMedium: 'referral',
  utmCampaign: 'tool-discovery',
  affiliateId: 'runatlas-001'
};

export function wrapAffiliateLink(url: string, toolName?: string, config = DEFAULT_CONFIG): string {
  // Build URL with UTM params and optional affiliate ID
  const u = new URL(url);
  u.searchParams.set('utm_source', config.utmSource);
  u.searchParams.set('utm_medium', config.utmMedium);
  u.searchParams.set('utm_campaign', config.utmCampaign);
  if (toolName) u.searchParams.set('utm_content', toolName.toLowerCase().replace(/\s+/g, '-'));
  if (config.affiliateId) u.searchParams.set('ref', config.affiliateId);
  return u.toString();
}

/**
 * Obfuscates an email address to prevent spam scraping.
 * Use in JSX: <a href={mailtoHref("hello@runatlas.com")}>Contact</a>
 */
export function mailtoHref(email: string): string {
  // Encode each character as HTML entity
  return "mailto:" + email
    .split("")
    .map((c) => `&#${c.charCodeAt(0)};`)
    .join("");
}

/**
 * Returns an obfuscated display of an email address.
 * "hello@runatlas.com" → "hello&#64;runatlas&#46;com"
 */
export function obfuscateEmail(email: string): string {
  return email
    .replace("@", "&#64;")
    .replace(/\./g, "&#46;");
}

import { useEffect, useRef } from 'react';
import { motion } from 'framer-motion';

export function useScrollReveal() {
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          entry.target.classList.add('animate-in');
          observer.unobserve(entry.target);
        }
      },
      { threshold: 0.1 }
    );

    if (ref.current) {
      observer.observe(ref.current);
    }

    return () => {
      if (ref.current) {
        observer.unobserve(ref.current);
      }
    };
  }, []);

  return ref;
}

import { useEffect, useRef } from 'react';

export function useParallax(depth = 0.5) {
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const handleScroll = () => {
      if (!ref.current) return;
      const scrollY = window.scrollY;
      ref.current.style.transform = `translateY(${scrollY * depth}px)`;
    };

    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, [depth]);

  return ref;
}

// ⭐ SEO Meta Tags Management
import React from "react";

interface SEOMeta {
  title: string;
  description: string;
  keywords?: string;
  canonical?: string;
  ogImage?: string;
  ogType?: string;
  twitterCard?: string;
  author?: string;
  robots?: string;
}

/**
 * ⭐ Utility für SEO Meta Tags
 * Ersetzt alle Meta Tags automatisch
 */
export function setSEOMeta({
  title,
  description,
  keywords = "runatlas, tools, productivity",
  canonical,
  ogImage = "https://runatlas.com/og-default.png",
  ogType = "website",
  twitterCard = "summary_large_image",
  author = "RunAtlas",
  robots = "index, follow",
}: SEOMeta) {
  // ⭐ Title
  document.title = `${title} | RunAtlas`;

  // ⭐ Meta Tags setzen/aktualisieren
  const setMetaTag = (name: string, content: string) => {
    let element = document.querySelector(`meta[name="${name}"]`) as HTMLMetaElement;
    if (!element) {
      element = document.createElement("meta");
      element.name = name;
      document.head.appendChild(element);
    }
    element.content = content;
  };

  const setPropertyTag = (property: string, content: string) => {
    let element = document.querySelector(
      `meta[property="${property}"]`
    ) as HTMLMetaElement;
    if (!element) {
      element = document.createElement("meta");
      element.setAttribute("property", property);
      document.head.appendChild(element);
    }
    element.content = content;
  };

  // Standard Meta Tags
  setMetaTag("description", description);
  setMetaTag("keywords", keywords);
  setMetaTag("author", author);
  setMetaTag("robots", robots);

  // OpenGraph Tags
  setPropertyTag("og:title", title);
  setPropertyTag("og:description", description);
  setPropertyTag("og:image", ogImage);
  setPropertyTag("og:type", ogType);
  setPropertyTag("og:url", window.location.href);

  // Twitter Card Tags
  setMetaTag("twitter:card", twitterCard);
  setMetaTag("twitter:title", title);
  setMetaTag("twitter:description", description);
  setMetaTag("twitter:image", ogImage);

  // Canonical URL
  if (canonical) {
    let link = document.querySelector('link[rel="canonical"]') as HTMLLinkElement;
    if (!link) {
      link = document.createElement("link");
      link.rel = "canonical";
      document.head.appendChild(link);
    }
    link.href = canonical;
  }
}

/**
 * ⭐ Hook für SEO Meta Tags (für React Components)
 */
export function usePageMeta(meta: SEOMeta) {
  React.useEffect(() => {
    setSEOMeta(meta);
  }, [meta]);
}

// ⭐ Schema.org strukturierte Daten
export function setStructuredData(data: Record<string, any>) {
  let script = document.querySelector(
    'script[type="application/ld+json"]'
  ) as HTMLScriptElement;
  if (!script) {
    script = document.createElement("script");
    script.type = "application/ld+json";
    document.head.appendChild(script);
  }
  script.textContent = JSON.stringify(data);
}

// Stack sharing via URL parameters (Base64 encoded)

export interface SharedStack {
  name: string;
  toolIds: string[];
  created: string;
}

export function encodeStack(stack: SharedStack): string {
  const json = JSON.stringify(stack);
  return btoa(encodeURIComponent(json));
}

export function decodeStack(encoded: string): SharedStack | null {
  try {
    const json = decodeURIComponent(atob(encoded));
    return JSON.parse(json);
  } catch { return null; }
}

export function generateShareUrl(stack: SharedStack): string {
  const base = typeof window !== 'undefined' ? window.location.origin : '';
  return `${base}/?stack=${encodeStack(stack)}`;
}

export function getStackFromUrl(): SharedStack | null {
  if (typeof window === 'undefined') return null;
  const params = new URLSearchParams(window.location.search);
  const encoded = params.get('stack');
  if (!encoded) return null;
  return decodeStack(encoded);
}

export async function shareStack(stack: SharedStack): Promise<boolean> {
  const url = generateShareUrl(stack);
  try {
    if (navigator.share) {
      await navigator.share({ title: `RunAtlas Stack: ${stack.name}`, text: `Mein AI-Tool Stack mit ${stack.toolIds.length} Tools`, url });
      return true;
    }
    await navigator.clipboard.writeText(url);
    return true;
  } catch { return false; }
}

  } catch (error) {
    console.error("OG Image Generierung fehlgeschlagen:", error);
    return null;
  }
}

export function generateOgImageUrl({
  title,
  description,
  author,
  image,
  color = "#6366f1",
}: OGImageParams): string {
  // ⭐ Option 1: Eigene API Route
  const params = new URLSearchParams({
    title,
    description: description || "RunAtlas",
    author: author || "RunAtlas",
    color,
    ...(image && { image }),
  });

  return `/api/og-image?${params.toString()}`;
}

# RunAtlas Landing Page

> **The Map of the AI Universe** — marketing / landing site for [RunAtlas](https://runatlas.com).

Built with **Vite 6**, **React 19**, **TypeScript**, **Tailwind CSS v4**, and **Framer Motion**.

## Quick Start

```bash
# Clone and install
cp .env.example .env    # fill in VITE_APP_URL and VITE_SITE_URL
npm install
npm run dev             # → http://localhost:5173
```

## Available Scripts

| Script           | Action                          |
|------------------|---------------------------------|
| `npm run dev`    | Start development server        |
| `npm run build`  | Production build (to `dist/`)   |
| `npm run preview`| Preview production build        |
| `npm run lint`   | Run ESLint                      |
| `npm run format` | Format with Prettier            |
