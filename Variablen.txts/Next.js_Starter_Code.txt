export async function getTools(opts?: {
  category?: string;
  search?: string;
  sort?: string;
  limit?: number;
}): Promise<Tool[]> {
  let q = supabase
    .from("tools")
    .select("*")
    .eq("status", "active")
    .order(opts?.sort || "atlas_score", { ascending: false })
    .limit(opts?.limit || 50);

  if (opts?.category && opts.category !== "All")
    q = q.eq("category", opts.category);
  if (opts?.search)
    q = q.ilike("name", `%${opts.search}%`);

  const { data, error } = await q;
  if (error) throw error;
  return data || [];
}

export async function getToolBySlug(slug: string): Promise<Tool | null> {
  const { data } = await supabase
    .from("tools")
    .select("*")
    .eq("slug", slug)
    .single();
  return data;
}

export async function getStacks(): Promise<Stack[]> {
  const { data } = await supabase
    .from("stacks")
    .select("*, steps:stack_steps(*)")
    .eq("is_public", true)
    .order("views", { ascending: false });
  return data || [];
}

export async function saveStack(stack: Omit<Stack, "id" | "created_at" | "views" | "clones">): Promise<Stack> {
  const { data, error } = await supabase
    .from("stacks")
    .insert(stack)
    .select()
    .single();
  if (error) throw error;
  return data;
}

      setErr(e.message || "Generation failed.");

import Anthropic from "@anthropic-ai/sdk";
import { NextRequest, NextResponse } from "next/server";

const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

export async function POST(req: NextRequest) {
  const { goal } = await req.json();
  if (!goal) return NextResponse.json({ error: "No goal provided" }, { status: 400 });

  const msg = await anthropic.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 1200,
    messages: [{
      role: "user",
      content: `You are ATLAS OS. User goal: "${goal}"
Return ONLY valid JSON:
{"title":"...","summary":"...","estimated_cost":"$X/mo","difficulty":"easy|medium|hard","time_to_result":"...","wsrScore":72,"steps":[{"step":1,"tool":"...","role":"...","action":"...","why":"...","altTool":"..."}],"pro_tip":"...","moat":"..."}`
    }]
  });

  const text = msg.content.find((b) => b.type === "text")?.text || "";
  try {
    const result = JSON.parse(text.replace(/```json|```/g, "").trim());
    return NextResponse.json(result);
  } catch {
    return NextResponse.json({ error: "Parse error" }, { status: 500 });
  }
}

import { getToolBySlug, getTools } from "@/lib/api";
import { notFound } from "next/navigation";
import type { Metadata } from "next";

export async function generateMetadata({ params }: { params: { slug: string } }): Promise<Metadata> {
  const tool = await getToolBySlug(params.slug);
  if (!tool) return {};
  return {
    title: `${tool.name} — ATLAS OS`,
    description: tool.description,
  };
}

export default async function ToolPage({ params }: { params: { slug: string } }) {
  const tool = await getToolBySlug(params.slug);
  if (!tool) notFound();

  return (
    <section className="max-w-[800px] mx-auto px-5 py-14">
      <div className="bg-[#0f172a] border border-[#1e293b] rounded-2xl p-8">
        <div className="flex gap-4 items-start mb-6">
          <span className="text-5xl">{tool.logo}</span>
          <div>
            <h1 className="text-2xl font-black text-slate-100">{tool.name}</h1>
            <p className="text-slate-500 text-sm mt-1">{tool.company}</p>
            <div className="flex gap-2 mt-2 flex-wrap">
              <span className="text-[10px] bg-cyan-400/10 text-cyan-400 border border-cyan-400/20 rounded px-2 py-0.5">
                {tool.category}
              </span>
              {tool.api_available && (

import { MetadataRoute } from "next";
import { getTools, getStacks } from "@/lib/api";

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const [tools, stacks] = await Promise.all([getTools({ limit: 10000 }), getStacks()]);
  const base = process.env.NEXT_PUBLIC_SITE_URL || "https://atlas-os.com";

  return [
    { url: base, changeFrequency: "daily", priority: 1 },
    { url: `${base}/generate`, changeFrequency: "weekly", priority: 0.9 },
    { url: `${base}/stacks`, changeFrequency: "daily", priority: 0.8 },
    { url: `${base}/tools`, changeFrequency: "daily", priority: 0.8 },
    { url: `${base}/compare`, changeFrequency: "weekly", priority: 0.7 },
    { url: `${base}/pricing`, changeFrequency: "monthly", priority: 0.6 },
    ...tools.map((t) => ({ url: `${base}/tools/${t.slug}`, changeFrequency: "weekly" as const, priority: 0.7 })),
    ...stacks.map((s) => ({ url: `${base}/stacks/${s.slug}`, changeFrequency: "weekly" as const, priority: 0.7 })),
  ];
}

import { MetadataRoute } from "next";
export default function robots(): MetadataRoute.Robots {
  return {
    rules: { userAgent: "*", allow: "/" },
    sitemap: `${process.env.NEXT_PUBLIC_SITE_URL}/sitemap.xml`,
  };
}
