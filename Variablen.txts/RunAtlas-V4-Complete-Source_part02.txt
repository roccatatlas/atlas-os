const DATA_DIR = path.join(process.cwd(), "data")
const TOOLS_PATH = path.join(DATA_DIR, "tools.json")
const COMPANIES_PATH = path.join(DATA_DIR, "companies.json")
const WORKFLOWS_PATH = path.join(DATA_DIR, "workflows.json")
const SUBMISSIONS_PATH = path.join(DATA_DIR, "submissions.json")
const USE_CASES_PATH = path.join(DATA_DIR, "use-cases.json")
const CAPABILITIES_PATH = path.join(DATA_DIR, "capabilities.json")

let toolsCache: Tool[] | null = null
let companiesCache: Company[] | null = null
let workflowsCache: Workflow[] | null = null
let capabilitiesCache: Capability[] | null = null

function readJson<T>(filePath: string, fallback: T): T {
  try {
    if (!fs.existsSync(filePath)) return fallback
    const raw = fs.readFileSync(filePath, "utf-8")
    return JSON.parse(raw) as T
  } catch (e) {
    console.error(`Failed to read ${filePath}:`, e)
    return fallback
  }
}

function writeJson(filePath: string, data: unknown): void {
  const dir = path.dirname(filePath)
  if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true })
  fs.writeFileSync(filePath, JSON.stringify(data, null, 2), "utf-8")
}

export function invalidateCache(): void {
  toolsCache = null
  companiesCache = null
  workflowsCache = null
  capabilitiesCache = null
}

// ============================================================
// Tools
// ============================================================

export function getTools(): Tool[] {
  if (toolsCache) return toolsCache
  toolsCache = readJson<Tool[]>(TOOLS_PATH, [])
  return toolsCache
}

export function getToolBySlug(slug: string): Tool | undefined {
  return getTools().find((t) => t.slug === slug)
}

export function getToolsByCategory(category: string): Tool[] {
  return getTools().filter((t) => t.category === category)
}

export function getToolsByCapability(capabilitySlug: string): Tool[] {
  return getTools().filter((t) =>
    t.capabilities?.includes(capabilitySlug)
  )
}

export function getToolsBySlugs(slugs: string[]): Tool[] {
  const slugSet = new Set(slugs)
  return getTools().filter((t) => slugSet.has(t.slug))
}

export function addTool(tool: Tool): void {
  const tools = getTools()
  tools.push(tool)
  toolsCache = tools
  writeJson(TOOLS_PATH, tools)
}

// ============================================================
// Companies
// ============================================================

export function getCompanies(): Company[] {
  if (companiesCache) return companiesCache
  companiesCache = readJson<Company[]>(COMPANIES_PATH, [])
  return companiesCache
}

export function getCompanyBySlug(slug: string): Company | undefined {
  return getCompanies().find((c) => c.slug === slug)
}

// ============================================================
// Workflows
// ============================================================

export function getWorkflows(): Workflow[] {
  if (workflowsCache) return workflowsCache
  workflowsCache = readJson<Workflow[]>(WORKFLOWS_PATH, [])
  return workflowsCache
}

export function getWorkflowBySlug(slug: string): Workflow | undefined {
  return getWorkflows().find((w) => w.slug === slug)
}

export function getWorkflowsForTool(toolSlug: string): Workflow[] {
  return getWorkflows().filter((w) =>
    w.steps.some((s) => s.tool_slug === toolSlug)
  )
}

// ============================================================
// Submissions
// ============================================================

export function getSubmissions(): Submission[] {
  return readJson<Submission[]>(SUBMISSIONS_PATH, [])
}

export function addSubmission(submission: Submission): void {
  const submissions = getSubmissions()
  submissions.push(submission)
  writeJson(SUBMISSIONS_PATH, submissions)
}

export function updateSubmissionStatus(
  id: string,
  status: "approved" | "rejected",
  notes?: string
): void {
  const submissions = getSubmissions()
  const idx = submissions.findIndex((s) => s.id === id)
  if (idx !== -1) {
    submissions[idx].status = status
    if (notes) submissions[idx].notes = notes
    writeJson(SUBMISSIONS_PATH, submissions)
  }
}

  return readJson<UseCase[]>(USE_CASES_PATH, [])
}

export function getStats() {
  const tools = getTools()
  const companies = getCompanies()
  const workflows = getWorkflows()
  const capabilities = getCapabilities()
  const categories = [...new Set(tools.map((t) => t.category))]

  return {
    totalTools: tools.length,
    totalCompanies: companies.length,
    totalWorkflows: workflows.length,
    totalCapabilities: capabilities.length,
    totalCategories: categories.length,
    freeTools: tools.filter((t) =>
      t.pricing === "free" || t.pricing === "freemium" || t.pricing_details?.free_tier
    ).length,
    openSourceTools: tools.filter((t) => t.is_open_source).length,
    apiTools: tools.filter((t) => t.has_api).length,
  }
}

/**

 *     "category": "llm",

 * The pipeline:
 * 1. Read source file (JSON or CSV)

 * 3. Deduplicate against existing tools

function slugify(text: string): string {
  return text
    .toLowerCase()
    .trim()
    .replace(/[^\w\s-]/g, "")
    .replace(/[\s_]+/g, "-")
    .replace(/-+/g, "-")
    .replace(/^-|-$/g, "")
}

const CATEGORY_KEYWORDS: Record<string, string[]> = {
  llm: ["language model", "chatbot", "gpt", "llm", "conversational", "chat ai"],
  image_generation: ["image", "art", "illustration", "picture", "photo generation"],
  video: ["video", "animation", "film", "movie", "clip generation"],
  audio: ["audio", "voice", "speech", "music", "sound", "tts", "podcast"],
  code_assistant: ["code", "coding", "programming", "developer", "ide", "copilot"],
  writing: ["writing", "content", "copywriting", "blog", "article", "text"],
  productivity: ["productivity", "workflow", "task", "project", "notes"],
  design: ["design", "ui", "ux", "figma", "graphic", "logo"],
  marketing: ["marketing", "seo", "ads", "advertising", "social media"],
  research: ["research", "academic", "papers", "science", "scholar"],
  automation: ["automation", "automate", "workflow", "no-code", "zapier"],
  data: ["data", "analytics", "database", "visualization", "chart"],
  education: ["education", "learning", "tutor", "study", "course"],
}

function detectCategory(tool: RawTool): string {
  if (tool.category) {
    const normalized = tool.category.toLowerCase().replace(/\s+/g, "_")
    if (Object.keys(CATEGORY_KEYWORDS).includes(normalized)) return normalized
  }

  const text = `${tool.name} ${tool.description || ""} ${(tool.tags || []).join(" ")}`.toLowerCase()

  let bestCategory = "other"
  let bestScore = 0

  for (const [category, keywords] of Object.entries(CATEGORY_KEYWORDS)) {
    const score = keywords.filter((kw) => text.includes(kw)).length
    if (score > bestScore) {
      bestScore = score
      bestCategory = category
    }
  }

  return bestCategory
}

// Capability Detection
// ============================================================

const CATEGORY_CAPABILITIES: Record<string, string[]> = {
  llm: ["text-generation", "chatbot-agents"],
  image_generation: ["image-generation", "design-creative"],
  video: ["video-generation"],
  audio: ["voice-speech"],
  code_assistant: ["code-generation"],
  writing: ["text-generation"],
  productivity: ["automation"],
  design: ["design-creative"],
  marketing: ["seo-marketing", "text-generation"],
  research: ["search-research"],
  automation: ["automation"],
  data: ["data-analysis"],
}

function detectCapabilities(category: string, tags: string[]): string[] {
  const caps = new Set(CATEGORY_CAPABILITIES[category] || [])
  const tagLower = tags.map((t) => t.toLowerCase())

  if (tagLower.some((t) => t.includes("music"))) caps.add("music-generation")
  if (tagLower.some((t) => t.includes("translation"))) caps.add("translation")
  if (tagLower.some((t) => t.includes("transcription") || t.includes("meeting"))) caps.add("meeting-transcription")
  if (tagLower.some((t) => t.includes("presentation"))) caps.add("presentation")

  return [...caps]
}

function normalizeTool(raw: RawTool): NormalizedTool {
  const category = detectCategory(raw)
  const tags = (raw.tags || []).map((t) => t.toLowerCase().trim())
  const capabilities = detectCapabilities(category, tags)
  const pricing = (raw.pricing || "freemium").toLowerCase()
  const isFree = pricing === "free" || pricing === "freemium"
  const isOSS = raw.is_open_source || tags.some((t) => t.includes("open-source") || t.includes("open source"))

  return {
    name: raw.name.trim(),
    slug: slugify(raw.name),
    description: (raw.description || `${raw.name} is an AI tool.`).trim(),
    long_description: (raw.long_description || raw.description || "").trim(),
    website: raw.website.trim(),
    logo: null,
    category,
    pricing,
    company: raw.company || "",
    company_slug: raw.company ? slugify(raw.company) : "",
    tags,
    rating: raw.rating ?? 4.0,
    features: raw.features || [],
    pricing_details: {
      free_tier: isFree,
      starting_price: "",
      enterprise: false,
    },
    founded_year: raw.founded_year || null,
    hq_country: raw.hq_country || "",
    has_api: raw.has_api ?? false,
    is_open_source: isOSS ?? false,
    platforms: raw.platforms || ["web"],
    capabilities,
  }
}

// ============================================================
// CSV Parser (simple — handles basic CSV)
// ============================================================

function parseCSV(content: string): RawTool[] {
  const lines = content.trim().split("\n")
  if (lines.length < 2) return []

  const headers = lines[0].split(",").map((h) => h.trim().replace(/"/g, ""))

  return lines.slice(1).map((line) => {
    const values = line.split(",").map((v) => v.trim().replace(/^"|"$/g, ""))
    const obj: Record<string, string> = {}
    headers.forEach((h, i) => {
      obj[h] = values[i] || ""
    })

    return {
      name: obj.name || obj.Name || obj.tool_name || "",
      website: obj.website || obj.url || obj.Website || "",
      description: obj.description || obj.Description || "",
      category: obj.category || obj.Category || "",
      pricing: obj.pricing || obj.Pricing || "",
      company: obj.company || obj.Company || "",
      tags: (obj.tags || obj.Tags || "").split(";").filter(Boolean),
    } as RawTool
  }).filter((t) => t.name && t.website)
}

async function importTools() {
  const args = process.argv.slice(2)
  const sourceIdx = args.indexOf("--source")
  const dryRun = args.includes("--dry-run")

  if (sourceIdx === -1 || !args[sourceIdx + 1]) {
    console.error("Usage: npx ts-node scripts/import-tools.ts --source <file> [--dry-run]")
    process.exit(1)
  }

  const sourceFile = args[sourceIdx + 1]
  const sourcePath = path.resolve(sourceFile)

  if (!fs.existsSync(sourcePath)) {
    console.error(`Source file not found: ${sourcePath}`)
    process.exit(1)
  }

  console.log(`📦 RunAtlas Import Pipeline`)
  console.log(`  Source: ${sourcePath}`)
  console.log(`  Dry run: ${dryRun}`)
  console.log("")

  // 1. Read source
  const content = fs.readFileSync(sourcePath, "utf-8")
  let rawTools: RawTool[]

  if (sourcePath.endsWith(".csv")) {
    rawTools = parseCSV(content)
    console.log(`  📄 Parsed ${rawTools.length} tools from CSV`)
  } else {
    rawTools = JSON.parse(content)
    if (!Array.isArray(rawTools)) {
      console.error("JSON source must be an array of tools")
      process.exit(1)
    }
    console.log(`  📄 Parsed ${rawTools.length} tools from JSON`)
  }

  // 2. Normalize
  const normalized = rawTools.map(normalizeTool)
  console.log(`  🔧 Normalized ${normalized.length} tools`)

  // 3. Deduplicate against existing
  const toolsPath = path.join(process.cwd(), "data", "tools.json")
  const existing: NormalizedTool[] = fs.existsSync(toolsPath)
    ? JSON.parse(fs.readFileSync(toolsPath, "utf-8"))
    : []

  const existingSlugs = new Set(existing.map((t) => t.slug))
  const existingWebsites = new Set(
    existing.map((t) => t.website.replace(/^https?:\/\//, "").replace(/\/$/, ""))
  )

  const newTools = normalized.filter((t) => {
    const websiteNorm = t.website.replace(/^https?:\/\//, "").replace(/\/$/, "")
    if (existingSlugs.has(t.slug)) return false
    if (existingWebsites.has(websiteNorm)) return false
    return true
  })

  const duplicates = normalized.length - newTools.length
  console.log(`  🔍 Deduplication: ${duplicates} duplicates skipped`)
  console.log(`  ✨ ${newTools.length} new tools to import`)

  if (dryRun) {
    console.log("\n🏃 DRY RUN — no changes written")
    console.log("\nSample new tools:")
    newTools.slice(0, 5).forEach((t) => {
      console.log(`  - ${t.name} (${t.category}) → ${t.website}`)
    })
    return
  }

  if (newTools.length === 0) {
    console.log("\n✅ No new tools to import — database is up to date")
    return
  }

  // 4. Write to tools.json
  const updated = [...existing, ...newTools]
  fs.writeFileSync(toolsPath, JSON.stringify(updated, null, 2), "utf-8")

  console.log(`\n✅ Import complete!`)
  console.log(`  Before: ${existing.length} tools`)
  console.log(`  Added: ${newTools.length} tools`)
  console.log(`  After: ${updated.length} tools`)

  // Category breakdown of new tools
  const catBreakdown: Record<string, number> = {}
  for (const t of newTools) {
    catBreakdown[t.category] = (catBreakdown[t.category] || 0) + 1
  }
  console.log("\n  New tools by category:")
  for (const [cat, count] of Object.entries(catBreakdown).sort((a, b) => b[1] - a[1])) {
    console.log(`    ${cat}: ${count}`)
  }
}

importTools().catch((e) => {
  console.error("Import failed:", e)
  process.exit(1)
})

 * - Normalizes slugs, categories, pricing
 * - Removes exact duplicates
 * - Validates URLs and ratings
 * - Reports data quality issues
 *
 * Usage:
 *   npx ts-node scripts/normalize.ts
 *   npx ts-node scripts/normalize.ts --fix     (auto-fix issues)
 *   npx ts-node scripts/normalize.ts --report   (report only, no changes)
 */

import fs from "fs"
import path from "path"

const TOOLS_PATH = path.join(process.cwd(), "data", "tools.json")
const args = process.argv.slice(2)
const fixMode = args.includes("--fix")
const reportOnly = args.includes("--report")

interface Tool {
  name: string
  slug: string
  description: string
  long_description: string
  website: string
  logo: string | null
  category: string
  pricing: string
  company: string
  company_slug: string
  tags: string[]
  rating: number
  features: string[]
  pricing_details: {
    free_tier: boolean
    starting_price: string
    enterprise: boolean
  }
  founded_year: number | null
  hq_country: string
  has_api: boolean
  is_open_source: boolean
  platforms: string[]
  capabilities: string[]
}

function slugify(text: string): string {
  return text
    .toLowerCase()
    .trim()
    .replace(/[^\w\s-]/g, "")
    .replace(/[\s_]+/g, "-")
    .replace(/-+/g, "-")
    .replace(/^-|-$/g, "")
}

const VALID_CATEGORIES = [
  "llm", "image_generation", "video", "audio", "code_assistant",
  "writing", "productivity", "design", "marketing", "research",
  "education", "data", "automation", "agents", "chatbot", "search", "other",
]

const VALID_PRICING = ["free", "freemium", "paid", "open_source", "enterprise"]

function normalize() {
  console.log("🔧 RunAtlas Data Normalizer")
  console.log(`  Mode: ${reportOnly ? "Report only" : fixMode ? "Auto-fix" : "Standard"}`)
  console.log("")

  const raw = fs.readFileSync(TOOLS_PATH, "utf-8")
  const tools: Tool[] = JSON.parse(raw)

  let issues = 0
  let fixed = 0

  // 1. Check for duplicates
  const slugMap = new Map<string, number>()
  const websiteMap = new Map<string, number>()

  tools.forEach((t, i) => {
    if (slugMap.has(t.slug)) {
      console.warn(`  ⚠️ Duplicate slug: "${t.slug}" at index ${i} (first at ${slugMap.get(t.slug)})`)
      issues++
    } else {
      slugMap.set(t.slug, i)
    }

    const normUrl = t.website.replace(/^https?:\/\//, "").replace(/\/$/, "").toLowerCase()
    if (websiteMap.has(normUrl)) {
      console.warn(`  ⚠️ Duplicate website: "${normUrl}" at index ${i}`)
      issues++
    } else {
      websiteMap.set(normUrl, i)
    }
  })

  // 2. Validate & normalize each tool
  for (const tool of tools) {
    // Fix slug
    const expectedSlug = slugify(tool.name)
    if (!tool.slug) {
      tool.slug = expectedSlug
      console.log(`  🔧 Generated slug for "${tool.name}": ${tool.slug}`)
      fixed++
    }

    // Validate category
    if (!VALID_CATEGORIES.includes(tool.category)) {
      console.warn(`  ⚠️ Unknown category "${tool.category}" for "${tool.name}"`)
      issues++
      if (fixMode) {
        tool.category = "other"
        fixed++
      }
    }

    // Validate pricing
    if (tool.pricing && !VALID_PRICING.includes(tool.pricing.toLowerCase())) {
      console.warn(`  ⚠️ Unknown pricing "${tool.pricing}" for "${tool.name}"`)
      issues++
      if (fixMode) {
        tool.pricing = "freemium"
        fixed++
      }
    }

    // Validate rating
    if (tool.rating < 0 || tool.rating > 5 || isNaN(tool.rating)) {
      console.warn(`  ⚠️ Invalid rating ${tool.rating} for "${tool.name}"`)
      issues++
      if (fixMode) {
        tool.rating = Math.max(0, Math.min(5, tool.rating || 4.0))
        fixed++
      }
    }

    // Validate URL
    if (!tool.website.startsWith("http")) {
      console.warn(`  ⚠️ Invalid URL "${tool.website}" for "${tool.name}"`)
      issues++
      if (fixMode) {
        tool.website = `https://${tool.website}`
        fixed++
      }
    }

    // Fill missing arrays
    if (!tool.tags) { tool.tags = []; fixed++ }
    if (!tool.features) { tool.features = []; fixed++ }
    if (!tool.platforms) { tool.platforms = ["web"]; fixed++ }
    if (!tool.capabilities) { tool.capabilities = []; fixed++ }

    // Fill missing booleans
    if (tool.has_api === undefined) tool.has_api = false
    if (tool.is_open_source === undefined) tool.is_open_source = false

    // Fill missing pricing_details
    if (!tool.pricing_details) {
      tool.pricing_details = {
        free_tier: tool.pricing === "free" || tool.pricing === "freemium",
        starting_price: "",
        enterprise: false,
      }
      fixed++
    }

    // Fill missing description
    if (!tool.description) {
      tool.description = `${tool.name} is an AI tool.`
      fixed++
    }
    if (!tool.long_description) {
      tool.long_description = tool.description
    }

    // Fill missing company info
    if (!tool.company) tool.company = ""
    if (!tool.company_slug) tool.company_slug = tool.company ? slugify(tool.company) : ""
    if (!tool.hq_country) tool.hq_country = ""
  }

  // 3. Remove exact duplicates (by slug)
  const uniqueTools: Tool[] = []
  const seenSlugs = new Set<string>()
  for (const tool of tools) {
    if (!seenSlugs.has(tool.slug)) {
      seenSlugs.add(tool.slug)
      uniqueTools.push(tool)
    }
  }

  const removed = tools.length - uniqueTools.length
  if (removed > 0) {
    console.log(`  🗑️ Removed ${removed} duplicates`)
  }

  // 4. Stats
  console.log("")
  console.log("📊 Data Quality Report:")
  console.log(`  Total tools: ${uniqueTools.length}`)
  console.log(`  Issues found: ${issues}`)
  console.log(`  Auto-fixed: ${fixed}`)
  console.log(`  Duplicates removed: ${removed}`)
  console.log("")

  // Category distribution
  const catDist: Record<string, number> = {}
  for (const t of uniqueTools) {
    catDist[t.category] = (catDist[t.category] || 0) + 1
  }
  console.log("  Category distribution:")
  for (const [cat, count] of Object.entries(catDist).sort((a, b) => b[1] - a[1])) {
    console.log(`    ${cat}: ${count}`)
  }

  // Missing data report
  const missingDesc = uniqueTools.filter((t) => t.description.length < 20).length
  const missingFeatures = uniqueTools.filter((t) => t.features.length === 0).length
  const missingTags = uniqueTools.filter((t) => t.tags.length === 0).length
  const missingCaps = uniqueTools.filter((t) => t.capabilities.length === 0).length

  console.log("")
  console.log("  Data completeness:")
  console.log(`    Short descriptions (< 20 chars): ${missingDesc}`)
  console.log(`    Missing features: ${missingFeatures}`)
  console.log(`    Missing tags: ${missingTags}`)
  console.log(`    Missing capabilities: ${missingCaps}`)

  // 5. Write if not report-only
  if (!reportOnly) {
    fs.writeFileSync(TOOLS_PATH, JSON.stringify(uniqueTools, null, 2), "utf-8")
    console.log(`\n✅ Written ${uniqueTools.length} normalized tools to ${TOOLS_PATH}`)
  } else {
    console.log("\n📋 Report only — no changes written")
  }
}

// ============================================================
// GitHub Scraper — finds AI tool repos
// ============================================================

async function scrapeGitHub(maxPages = 5): Promise<ScrapedTool[]> {
  const tools: ScrapedTool[] = []
  const queries = [
    "ai+tool+stars:>100",
    "llm+framework+stars:>50",
    "image+generation+ai+stars:>100",
    "voice+ai+stars:>50",
    "ai+agent+stars:>100",
    "text-to-speech+stars:>50",
  ]

  for (const query of queries) {
    try {
      const url = `https://api.github.com/search/repositories?q=${query}&sort=stars&per_page=30`
      const headers: Record<string, string> = {
        Accept: "application/vnd.github.v3+json",
        "User-Agent": "RunAtlas-Scraper",
      }
      if (process.env.GITHUB_TOKEN) {
        headers.Authorization = `token ${process.env.GITHUB_TOKEN}`
      }

      const res = await fetch(url, { headers })
      if (!res.ok) {
        console.warn(`  ⚠️ GitHub API error for "${query}": ${res.status}`)
        continue
      }

      const data = (await res.json()) as { items: Array<{
        name: string
        html_url: string
        description: string
        topics: string[]
        homepage: string
        stargazers_count: number
      }> }

      for (const repo of data.items) {
        if (repo.stargazers_count < 50) continue

        tools.push({
          name: repo.name
            .replace(/-/g, " ")
            .replace(/\b\w/g, (c) => c.toUpperCase()),
          website: repo.homepage || repo.html_url,
          description: repo.description || "",
          tags: repo.topics?.slice(0, 5) || [],
          pricing: "open_source",
          source: "github",
          scraped_at: new Date().toISOString(),
        })
      }

      console.log(`  ✓ GitHub "${query}": ${data.items.length} repos`)

      // Rate limiting
      await new Promise((r) => setTimeout(r, 1000))
    } catch (e) {
      console.warn(`  ⚠️ GitHub scrape failed for "${query}":`, e)
    }
  }

  return tools
}

  if (!PH_TOKEN) {

  // Product Hunt GraphQL API
  const query = `{
    posts(topic: "artificial-intelligence", first: 50, order: VOTES) {

        node {

          tagline
          url
          website
          topics { edges { node { name } } }
          votesCount
        }
      }
    }
  }`

  try {
    const res = await fetch("https://api.producthunt.com/v2/api/graphql", {
      method: "POST",
      headers: {
        Authorization: `Bearer ${PH_TOKEN}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ query }),
    })

    if (!res.ok) {
      console.warn(`  ⚠️ Product Hunt API error: ${res.status}`)
      return []
    }

    const data = (await res.json()) as { data: { posts: { edges: Array<{
      node: {
        name: string
        tagline: string
        url: string
        website: string
        topics: { edges: Array<{ node: { name: string } }> }
        votesCount: number
      }
    }> } } }

    return data.data.posts.edges.map((e) => ({
      name: e.node.name,
      website: e.node.website || e.node.url,
      description: e.node.tagline,
      tags: e.node.topics.edges.map((t) => t.node.name.toLowerCase()),
      source: "producthunt",
      scraped_at: new Date().toISOString(),

  } catch (e) {

        <div className="mt-8 bg-surface border border-gold/5 rounded-sm p-6">
          <h3 className="font-mono text-xs uppercase tracking-[0.2em] text-gold mb-6">
            <Zap size={12} className="inline mr-2" />
            Free Tools Available
          </h3>
          <div className="flex items-center gap-8">
            <div>
              <p className="font-display text-5xl text-paper">{stats.freeTools}</p>
              <p className="font-mono text-xs text-mist/60 mt-1">Free & Open Source tools</p>
            </div>
            <div className="flex-grow h-4 bg-paper/5 rounded-full overflow-hidden">
              <div
                className="h-full bg-gradient-to-r from-teal to-gold rounded-full"
                style={{ width: `${Math.round((stats.freeTools / stats.totalTools) * 100)}%` }}
              />
            </div>
            <div>
              <p className="font-display text-2xl text-mist">{Math.round((stats.freeTools / stats.totalTools) * 100)}%</p>
              <p className="font-mono text-xs text-mist/60 mt-1">of all tools</p>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
