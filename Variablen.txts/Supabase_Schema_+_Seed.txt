-- ── EXTENSIONS ───────────────────────────────────────────────
create extension if not exists "uuid-ossp";
create extension if not exists "pg_trgm";

-- ── CLEANUP (safe re-run) ─────────────────────────────────────
drop table if exists tool_comparisons  cascade;
drop table if exists stack_steps       cascade;
drop table if exists stacks            cascade;
drop table if exists tool_integrations cascade;
drop table if exists tool_categories   cascade;
drop table if exists tools             cascade;
drop table if exists categories        cascade;
drop table if exists seo_pages         cascade;
drop table if exists slug_history      cascade;
drop table if exists page_analytics    cascade;

-- ── CATEGORIES ───────────────────────────────────────────────
create table categories (
  id           serial primary key,
  slug         text unique not null,
  name         text not null,
  super_cat    text,
  description  text,
  tool_count   int not null default 0,
  icon         text,
  display_order int default 999,
  status       text default 'active',
  created_at   timestamptz default now()
);

-- ── TOOLS ────────────────────────────────────────────────────
create table tools (
  id               uuid primary key default uuid_generate_v4(),
  slug             text unique not null,
  name             text not null,
  description      text not null,
  category         text not null,
  company          text,
  pricing          text,
  pricing_num      decimal(10,2) default 0,
  logo             text default '🤖',
  rating           decimal(3,2) default 4.0,
  users            text default '—',
  api_available    boolean default false,
  open_source      boolean default false,
  deploy           text default 'cloud',
  integrations     int default 0,
  context_window   text default '—',
  affiliate_url    text,
  atlas_score      decimal(5,2) default 50.0,
  monthly_visitors bigint,
  github_stars     int,
  website_url      text,
  sponsored_tier   text default 'none',
  tags             text[] default '{}',
  use_cases        text[] default '{}',
  status           text default 'active',
  created_at       timestamptz default now(),
  updated_at       timestamptz default now()
);

create index idx_tools_category    on tools (category);
create index idx_tools_atlas_score on tools (atlas_score desc);
create index idx_tools_slug        on tools (slug);
create index idx_tools_status      on tools (status);
create index idx_tools_tags        on tools using gin (tags);
create index idx_tools_name_trgm   on tools using gin (name gin_trgm_ops);

-- ── TOOL ↔ CATEGORY (join) ───────────────────────────────────
create table tool_categories (
  tool_id     uuid not null references tools(id) on delete cascade,
  category_id int  not null references categories(id) on delete cascade,
  primary key (tool_id, category_id)
);

-- ── TOOL INTEGRATIONS (knowledge graph edges) ────────────────
create table tool_integrations (
  tool_id            uuid not null references tools(id) on delete cascade,
  integrates_with_id uuid not null references tools(id) on delete cascade,
  created_at         timestamptz default now(),
  primary key (tool_id, integrates_with_id),
  constraint no_self_integration check (tool_id <> integrates_with_id)
);

create index idx_ti_tool_id  on tool_integrations (tool_id);
create index idx_ti_with_id  on tool_integrations (integrates_with_id);

-- ── STACKS ───────────────────────────────────────────────────
create table stacks (
  id           uuid primary key default uuid_generate_v4(),
  slug         text unique not null,
  title        text not null,
  goal         text,
  description  text,
  difficulty   text not null default 'medium'
               check (difficulty in ('easy','medium','hard')),
  cost_monthly text,
  views        int not null default 0,
  clones       int not null default 0,
  is_public    boolean not null default true,
  status       text default 'approved',
  created_at   timestamptz default now(),
  updated_at   timestamptz default now()
);

create index idx_stacks_slug      on stacks (slug);
create index idx_stacks_is_public on stacks (is_public);

-- ── STACK STEPS ──────────────────────────────────────────────
create table stack_steps (
  id          uuid primary key default uuid_generate_v4(),
  stack_id    uuid not null references stacks(id) on delete cascade,
  tool_name   text not null,
  role        text not null,
  action      text,
  step_order  int not null,
  constraint stack_steps_order_pos check (step_order >= 0),
  unique (stack_id, step_order)
);

create index idx_ss_stack_id on stack_steps (stack_id);

-- ── TOOL COMPARISONS ─────────────────────────────────────────
create table tool_comparisons (
  id           uuid primary key default uuid_generate_v4(),
  tool_a_id    uuid references tools(id) on delete cascade,
  tool_b_id    uuid references tools(id) on delete cascade,
  slug         text unique not null,
  verdict      text,
  content_json jsonb,
  views        int default 0,
  created_at   timestamptz default now(),
  unique (tool_a_id, tool_b_id)
);

-- ── SEO PAGES ────────────────────────────────────────────────
create table seo_pages (
  id               uuid primary key default uuid_generate_v4(),
  path             text unique not null,
  page_type        text not null,
  indexing_state   text not null default 'noindex'
                   check (indexing_state in ('indexable','noindex','blocked')),
  quality_passed   boolean default false,
  canonical_url    text,
  source_entity_id uuid,
  created_at       timestamptz default now(),
  updated_at       timestamptz default now()
);

-- ── SLUG HISTORY (SEO safety — 301s) ─────────────────────────
create table slug_history (
  id          uuid primary key default uuid_generate_v4(),
  entity_type text not null,
  entity_id   uuid not null,
  old_slug    text not null,
  new_slug    text not null,
  created_at  timestamptz default now(),
  unique (entity_type, old_slug)
);

-- ── PAGE ANALYTICS ───────────────────────────────────────────
create table page_analytics (
  id          bigserial primary key,
  page_url    text not null,
  page_type   text,
  views       int default 0,
  clicks      int default 0,
  conversions int default 0,
  recorded_at date default current_date
);

create index idx_pa_page_url on page_analytics (page_url);
create index idx_pa_date     on page_analytics (recorded_at desc);

-- ── AUTO updated_at TRIGGER ───────────────────────────────────
create or replace function set_updated_at()
returns trigger language plpgsql as $$
begin new.updated_at = now(); return new; end; $$;

create trigger tools_updated_at  before update on tools  for each row execute function set_updated_at();
create trigger stacks_updated_at before update on stacks for each row execute function set_updated_at();

-- ── RLS (Row Level Security) ──────────────────────────────────
alter table tools      enable row level security;
alter table stacks     enable row level security;
alter table stack_steps enable row level security;
alter table categories  enable row level security;

-- Public read access
create policy "Public read tools"       on tools       for select using (status = 'active');
create policy "Public read stacks"      on stacks      for select using (is_public = true);
create policy "Public read stack_steps" on stack_steps for select using (true);
create policy "Public read categories"  on categories  for select using (status = 'active');

-- ── COMPARISON SEED ──────────────────────────────────────────
insert into tool_comparisons (tool_a_id, tool_b_id, slug, verdict, views)
select a.id, b.id, a.slug || '-vs-' || b.slug,
  case
    when a.slug = 'chatgpt' and b.slug = 'claude'
      then 'Claude wins on reasoning and context window. ChatGPT wins on plugins and image generation.'
    when a.slug = 'midjourney' and b.slug = 'dalle-3'
      then 'Midjourney wins on artistic quality. DALL·E 3 wins on prompt accuracy and ChatGPT integration.'
    when a.slug = 'cursor' and b.slug = 'github-copilot'
      then 'Cursor wins on AI-native editing experience. GitHub Copilot wins on ecosystem integration.'
    else 'See full comparison for details.'
  end,
  floor(random() * 2000 + 500)::int
from tools a, tools b
where (a.slug, b.slug) in (
  ('chatgpt',    'claude'),
  ('midjourney', 'dalle-3'),
  ('cursor',     'github-copilot'),
  ('zapier',     'make'),
  ('elevenlabs', 'murf'),
  ('runway',     'pika'),
  ('notion-ai',  'obsidian'),
  ('perplexity', 'consensus')
);

-- ── UPDATE CATEGORY TOOL COUNTS ──────────────────────────────
update categories c
set tool_count = (
  select count(*) from tools t
  where lower(t.category) = lower(c.name)
  and t.status = 'active'
);
