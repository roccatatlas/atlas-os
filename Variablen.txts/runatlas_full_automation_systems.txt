import requests from bs4 import BeautifulSoup

def scrape_tool(url):

    html = requests.get(url).text
    soup = BeautifulSoup(html,"html.parser")

    title = soup.title.string if soup.title else "Unknown Tool"

    description = ""

    desc_tag = soup.find("meta",attrs={"name":"description"})

    if desc_tag:
        description = desc_tag.get("content")

    tool = {
        "name":title,
        "description":description,
        "website":url
    }

    return tool

def discover_tools():

    source = "https://www.producthunt.com/topics/artificial-intelligence"

    html = requests.get(source).text
    soup = BeautifulSoup(html,"html.parser")

    links = []

    for a in soup.find_all("a"):
        href = a.get("href")

        if href and "producthunt.com/posts" in href:
            links.append("https://producthunt.com"+href)

    return links

def run():

    urls = discover_tools()

    for url in urls:

        tool = scrape_tool(url)

        print("Discovered:",tool)

        save_to_database(tool)

run()
